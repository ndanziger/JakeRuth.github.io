001package com.commercehub.grails.shiro.guard;
002
003import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
004
005public class GuardArtefactHandler extends ArtefactHandlerAdapter {
006    public static final String TYPE = "Guard";
007
008    public GuardArtefactHandler() {
009        super(TYPE, GrailsGuardClass.class, DefaultGrailsGuardClass.class, null);
010    }
011
012    public boolean isArtefactClass(Class clazz) {
013        return clazz != null && clazz.getName().endsWith(TYPE);
014    }
015}
